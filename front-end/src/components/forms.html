            <!-- FORMULÁRIO AVANÇADO -->
          
            <!-- Select com Busca -->
            <div class="form-group">
              <label for="searchableSelect">Escolha uma opção:</label>
              <div class="custom-select">
                <!-- Campo de busca para filtrar opções -->
                <input
                  type="text"
                  id="selectSearchInput"
                  class="input-field"
                  placeholder="Buscar..."
                  oninput="filterSelectOptions()"
                />
                <!-- Lista de opções para o select -->
                <div class="options-list" id="selectOptions">
                  <div class="option" onclick="selectOption('Opção 1')">Opção 1</div>
                  <div class="option" onclick="selectOption('Opção 2')">Opção 2</div>
                  <div class="option" onclick="selectOption('Opção 3')">Opção 3</div>
                </div>
              </div>
            </div>
          
            <!-- Campo de Data com Calendário -->
            <div class="form-group">
              <label for="datePicker">Escolha uma data:</label>
              <!-- Campo de data com input do tipo date -->
              <input type="date" id="datePicker" class="input-field" />
            </div>
          
            <!-- Upload de Arquivos com Drag-and-Drop -->
            <div class="form-group">
              <label for="fileUpload">Upload de Arquivo:</label>
              <!-- Zona de drag-and-drop para upload de arquivos -->
              <div
                id="fileDropZone"
                class="file-dropzone"
                ondragover="handleDragOver(event)"
                ondrop="handleFileDrop(event)"
              >
                Arraste e solte arquivos aqui ou clique para fazer upload
                <!-- Input de arquivos escondido, ativado pelo clique -->
                <input
                  type="file"
                  id="fileInput"
                  class="hidden"
                  onchange="handleFileChange(event)"
                />
              </div>
              <!-- Lista de arquivos exibidos após o upload -->
              <div id="fileList" class="file-list"></div>
            </div>
          
            <!-- Botão de Submissão -->
            <button type="submit" class="btn primary">Enviar</button>
