      <!-- Este botão abre o modal pequeno -->
      <button class="btn primary" onclick="openModal('small-modal')">
        Abrir Modal Pequeno
      </button>
  
      <!-- Estrutura do Modal Pequeno -->
      <!-- Adicione um ID exclusivo e a classe 'small' para este modal -->
      <div id="small-modal" class="modal small">
        <div class="modal-content">
          <h3 class="modal-header">Modal Pequeno</h3>
          <p class="modal-body">Conteúdo do modal pequeno.</p>
          <div class="modal-footer">
            <!-- Botão para fechar o modal pequeno -->
            <button class="btn outlined" onclick="closeModal('small-modal')">
              Fechar
            </button>
          </div>
        </div>
      </div>

      <!-- Este botão abre um modal que contém um formulário -->
      <button class="btn outlined" onclick="openModal('form-modal')">
        Abrir Modal com Formulário
      </button>
  
      <!-- Estrutura do Modal com Formulário -->
      <!-- Adicione um ID exclusivo e uma classe, por exemplo 'medium', para este modal -->
      <div id="form-modal" class="modal medium">
        <div class="modal-content">
          <h3 class="modal-header">Formulário</h3>
          <!-- Corpo do modal com campos de formulário -->
          <form class="modal-body">
            <label for="name">Nome:</label>
            <!-- Campo para entrada do nome -->
            <input
              type="text"
              id="name"
              name="name"
              class="input-field"
              placeholder="Digite seu nome"
            />
  
            <label for="email">Email:</label>
            <!-- Campo para entrada do email -->
            <input
              type="email"
              id="email"
              name="email"
              class="input-field"
              placeholder="Digite seu email"
            />
          </form>
          <div class="modal-footer">
            <!-- Botão para cancelar e fechar o modal com formulário -->
            <button class="btn outlined" onclick="closeModal('form-modal')">
              Cancelar
            </button>
            <!-- Botão para submeter o formulário -->
            <button class="btn primary">Enviar</button>
          </div>
        </div>
      </div>
