Alerts e Notifications

Exemplos de alertas e notificações para diferentes cenários.

        <!-- Botões para mostrar alertas -->
        <div class="alerts-demo">
          <button class="btn primary" onclick="showAlert('success', 'Operação realizada com sucesso!')">
            Sucesso
          </button>
          <button class="btn outlined" onclick="showAlert('error', 'Ocorreu um erro ao realizar a operação.')">
            Erro
          </button>
          <button class="btn primary" onclick="showAlert('warning', 'Este é um aviso importante.')">
            Aviso
          </button>
          <button class="btn outlined" onclick="showAlert('info', 'Informação útil para o usuário.')">
            Informação
          </button>
        </div>
  
        <!-- Container de Alertas -->
        <div id="alertContainer" class="alert-container"></div>
  
        <!-- Script para mostrar alertas -->
        <script>
          function showAlert(type, message) {
            const alertContainer = document.getElementById("alertContainer");
            const alert = document.createElement("div");
            alert.className = \`alert alert-\${type}\`;
            alert.innerText = message;
  
            alertContainer.appendChild(alert);
  
            // Remover após 5 segundos
            setTimeout(() => {
              alert.remove();
            }, 5000);
          }
        </script>
